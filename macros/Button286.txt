$${

TOGGLE(@sprint);
    IF(@sprint);
        SETLABEL("StatusSPRINT","&6Sprint: &2ON");
        SET(@&info_msg,"&aSprint ativado.");
    ELSE;
        SETLABEL("StatusSPRINT","&6Sprint: &4OFF");

    ENDIF;
